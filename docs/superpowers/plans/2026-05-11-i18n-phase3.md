**Goal:** Admin UI (`app/`) を `next-intl` で完全に i18n 化し、JA + EN の 2 言語で動作可能にする。Phase 2b で導入した API code ベース訳出 (`app/src/lib/api-errors.ts`) をそのまま catalog に統合し、Phase 2 以前から残っていた UI 直書き日本語をすべて catalog へ移管する。

import { describe, it, expect } from "vitest";

import { NAMESPACES, LOCALES } from "@/i18n/config";

/**
 * 全 namespace で en と ja の key set が完全一致することを assert。
 * 翻訳忘れ / 余分なキーを CI で検出する。
 */
function flatKeys(obj: unknown, prefix = ""): string[] {
  if (!obj || typeof obj !== "object") return [];
  return Object.entries(obj as Record<string, unknown>).flatMap(([k, v]) => {
    const key = prefix ? `${prefix}.${k}` : k;
    if (v && typeof v === "object" && !Array.isArray(v)) {
      return flatKeys(v, key);
    }
    return [key];
  });
}

describe("locale keysets", () => {
  for (const ns of NAMESPACES) {
    it(`${ns}: en と ja の keys が一致`, async () => {
      const en = (await import(`@/locales/en/${ns}.json`)).default;
      const ja = (await import(`@/locales/ja/${ns}.json`)).default;
      const enKeys = flatKeys(en).sort();
      const jaKeys = flatKeys(ja).sort();
      expect(jaKeys).toEqual(enKeys);
    });
  }
});

```typescript
import { LOCALES, DEFAULT_LOCALE, FALLBACK_LOCALE_FOR_ACCEPT_LANGUAGE } from "@/i18n/config";
import { LOCALE_COOKIE_NAME, LOCALE_COOKIE_MAX_AGE } from "@/i18n/locale-cookie";

function resolveLocale(request: NextRequest): string {
  // 1) 既存 cookie 優先
  const existing = request.cookies.get(LOCALE_COOKIE_NAME)?.value;
  if (existing && (LOCALES as readonly string[]).includes(existing)) {
    return existing;
  }

  // 2) Accept-Language の最初の locale が LOCALES に含まれれば使う
  const acceptLanguage = request.headers.get("accept-language") ?? "";
  const candidates = acceptLanguage
    .split(",")
    .map((part) => part.split(";")[0].trim().split("-")[0].toLowerCase());
  for (const c of candidates) {
    if ((LOCALES as readonly string[]).includes(c)) {
      // 既存日本語ユーザー保護: cookie 未設定で Accept-Language に ja があれば ja
      return c;
    }
  }

  // 3) fallback (戦略 spec の「既存日本語ユーザー保護」のため一旦 ja)
  return FALLBACK_LOCALE_FOR_ACCEPT_LANGUAGE;
}

// 既存 proxy() の戻り直前で cookie をセット
export function proxy(request: NextRequest) {
  // ... 既存処理で response を作る
  const response = ... ; // 既存 logic

  // locale 未設定なら resolveLocale() の結果を cookie に書く
  if (!request.cookies.get(LOCALE_COOKIE_NAME)) {
    response.cookies.set(LOCALE_COOKIE_NAME, resolveLocale(request), {
      maxAge: LOCALE_COOKIE_MAX_AGE,
      path: "/",
      sameSite: "lax",
    });
  }
  return response;
}
```

- [ ] **Step 1**: hook 実装

```tsx
// app/src/hooks/use-locale-switcher.ts
"use client";
import { useRouter } from "next/navigation";
import { useTransition } from "react";
import { authClient } from "@/lib/auth/client";
import { LOCALE_COOKIE_NAME, LOCALE_COOKIE_MAX_AGE } from "@/i18n/locale-cookie";
import type { Locale } from "@/i18n/config";

export function useLocaleSwitcher() {
  const router = useRouter();
  const [isPending, startTransition] = useTransition();

  const switchTo = (locale: Locale) => {
    startTransition(async () => {
      // 1. cookie を先に書く (next-intl/request.ts が次の SSR で読む)
      document.cookie = `${LOCALE_COOKIE_NAME}=${locale}; max-age=${LOCALE_COOKIE_MAX_AGE}; path=/; samesite=lax`;
      // 2. DB に永続化 (ログイン中のみ。失敗しても cookie だけで動作するので silent)
      try {
        await authClient.setPreferredLocale(locale);
      } catch (err) {
        console.warn("setPreferredLocale failed (cookie still saved):", err);
      }
      // 3. RSC を再評価して翻訳を反映
      router.refresh();
    });
  };

  return { switchTo, isPending };
}
```

          size="sm"
          aria-label={t("switch")}
          disabled={isPending}
          data-testid="language-switcher"
        >
          <Globe className="h-4 w-4" />
          <span className="ml-2 text-xs uppercase">{current}</span>
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end">
        <DropdownMenuItem
          onClick={() => switchTo("en")}
          data-testid="language-switcher-en"
          aria-current={current === "en"}
        >
          {t("label_en")}
        </DropdownMenuItem>
        <DropdownMenuItem
          onClick={() => switchTo("ja")}
          data-testid="language-switcher-ja"
          aria-current={current === "ja"}
        >
          {t("label_ja")}
        </DropdownMenuItem>
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
