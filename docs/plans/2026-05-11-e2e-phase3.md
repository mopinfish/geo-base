### Step 3: AUTH-06 を実装

`app/tests/e2e/auth/auth-token-refresh.spec.ts` を新規作成:

```typescript
import { test, expect } from "../fixtures/authenticated-test";

import { resetDatabase } from "../utils/reset-db";
import { loginAsAdmin } from "../utils/session";
import { createTileset } from "../fixtures/factories";

/**
 * AUTH-06: access token 期限切れ → 自動 refresh で透明に継続。
 *
 * 設計:
 * - access_token_ttl_seconds = 900 (= 15 min)
 * - authClient は 60s 前にスケジュール refresh する仕組みなので、TTL 直後ではなく
 *   60s より手前で発火させたい。
 * - Playwright の `page.clock.install()` で時間を 16 分進めれば、ブラウザの
 *   タイマーが先に走り refresh が実行される。
 * - refresh 後も認証必須の API call (/api/tilesets) が成功することを assert。
 */
test.describe("Auth token refresh", () => {
  test.beforeAll(async () => {
    await loginAsAdmin();
    await resetDatabase();
  });

  test("AUTH-06 access token 期限切れで自動 refresh される", async ({ page }) => {
    // pre-condition: tileset が 1 件 → /tilesets で row が見える
    await createTileset({ name: "auth-06-precheck", type: "vector", isPublic: true });
    await page.goto("/tilesets");
    await expect(page.getByTestId("tileset-list-row")).toHaveCount(1, {
      timeout: 10_000,
    });

    // 時間を 16 分先に進める。authClient.scheduleRefresh が先に発火する想定。
    await page.clock.install({ time: new Date() });
    await page.clock.fastForward("16:00");

    // 再度 listings を読みに行く。期限切れの古い access_token で 401 → refresh →
    // retry → 200 → row 再表示 までを authClient が自動で処理する。
    await page.reload();
    await expect(page.getByTestId("tileset-list-row")).toHaveCount(1, {
      timeout: 15_000,
    });
  });
});
```

注: `page.clock` の挙動は Playwright のバージョンに依存。`@playwright/test` 1.45+ で使える。バージョンを確認:

```fish
grep '"@playwright/test"' app/package.json
```

1.45 未満なら `clock.fastForward` の代わりに `clock.runFor("16m")` を使うか、別アプローチ (`waitForTimeout(60_000)` で 60 秒だけ進める軽量版) に切り替える。

### Step 5: AK-07 (期限切れ key)

`app/tests/e2e/fixtures/factories.ts` の `createApiKey` シグネチャを拡張して `expiresAt` (ISO 文字列) を渡せるようにする。

```typescript
export async function createApiKey(input: {
  name: string;
  scopes?: string[];
  expiresAt?: string;  // ← 追加
}): Promise<CreatedApiKey> {
  const ctx = await createApiClient();
  try {
    const data: Record<string, unknown> = {
      name: input.name,
      scopes: input.scopes ?? ["read"],
    };
    if (input.expiresAt) {
      data.expires_at = input.expiresAt;
    }
    const res = await ctx.post("/api/api-keys", { data });
    if (!res.ok()) {
      throw new Error(`createApiKey failed: ${res.status()} ${await res.text()}`);
    }
    return (await res.json()) as CreatedApiKey;
  } finally {
    await ctx.dispose();
  }
}
```

実装の `ApiKeyCreate` モデルが `expires_at` を受けるかを確認:
```fish
grep -A 10 'class ApiKeyCreate' api/lib/models/api_key.py
```

期限切れキーを作るには `expires_at` を過去日 (例: `"2025-01-01T00:00:00Z"`) で投入する。API 側で「過去日は拒否」していたら、factory ではなく直接 SQL UPDATE で `expires_at` を過去化する経路に切り替える（その場合は `POST /api/test/expire-api-key` を test_helpers に追加）。

```typescript
 * minor / moderate は warning として log 出力するだけにし、初期導入時に
 * 既存ページが全部赤くなって CI が止まる事態を避ける。閾値は徐々に厳しくする。
 */
import { AxeBuilder } from "@axe-core/playwright";
import { expect, type Page } from "@playwright/test";

export async function expectNoA11ySeriousViolations(page: Page): Promise<void> {
  const results = await new AxeBuilder({ page })
    // ベスト共通の 4 タグ。重要度が低い practices は外す。
    .withTags(["wcag2a", "wcag2aa", "wcag21a", "wcag21aa"])
    .analyze();

  const severeViolations = results.violations.filter(
    (v) => v.impact === "serious" || v.impact === "critical",
  );

  if (severeViolations.length > 0) {
    // テストレポートに違反内容を出力。
    console.log(
      JSON.stringify(
        severeViolations.map((v) => ({
          id: v.id,
          impact: v.impact,
          help: v.help,
          nodes: v.nodes.length,
        })),
        null,
        2,
      ),
    );
  }

  expect(
    severeViolations,
    `Found ${severeViolations.length} serious/critical A11y violations`,
  ).toHaveLength(0);
}
```

### Step 1: flaky-summary.sh を「件数 + 率」を出すように拡張

`.github/workflows/scripts/flaky-summary.sh` を以下で置換:

```bash
#!/bin/bash
# Playwright が test-results/ に書き出す retry-N ディレクトリ数から flaky 件数を集計し、
# GitHub Actions の Job Summary に書き込む。Phase 3 拡張: 総テスト数で割って
# flake 率も出すことで、目標値 (< 2%) との比較がしやすくなる。
set -euo pipefail

SUMMARY="${GITHUB_STEP_SUMMARY:-/dev/stderr}"

echo "## Flaky tests summary" >> "$SUMMARY"

RESULTS_DIR="app/test-results"
if [ ! -d "$RESULTS_DIR" ]; then
  echo "No test-results directory at $RESULTS_DIR" >> "$SUMMARY"
  exit 0
fi

# 命名規則: <test-id>-retry1, <test-id>-retry2, ...
flaky_count=$(find "$RESULTS_DIR" -type d -name '*-retry*' | wc -l | tr -d ' ')

# Playwright の json reporter が --reporter=json で出力すれば総テスト数も
# 取れるが、現状は単に行数で代用する (各テストの自分自身のディレクトリ)。
total_count=$(find "$RESULTS_DIR" -maxdepth 1 -mindepth 1 -type d | wc -l | tr -d ' ')

echo "Total test attempts: $total_count" >> "$SUMMARY"
echo "Retry attempts (= flaky pass + final fail): $flaky_count" >> "$SUMMARY"

if [ "$total_count" -gt 0 ] && [ "$flaky_count" -gt 0 ]; then
  # bash で float 計算は dc 経由。
  rate=$(echo "scale=2; $flaky_count * 100 / $total_count" | bc)
  echo "Approximate flake rate: ${rate}% (target: < 2%)" >> "$SUMMARY"
fi

if [ "$flaky_count" -gt 0 ]; then
  echo "" >> "$SUMMARY"
  echo "Affected tests:" >> "$SUMMARY"
  find "$RESULTS_DIR" -type d -name '*-retry*' -print | sed 's|^|- |' >> "$SUMMARY"
fi
```

```python
_RECENT_PASSWORD_RESET_TOKENS: dict[str, str] = {}


def _record_password_reset_token(email: str, reset_url: str) -> None:
    """`reset_url` から token クエリを抽出して保持する。"""
    from urllib.parse import urlparse, parse_qs

    parsed = urlparse(reset_url)
    qs = parse_qs(parsed.query)
    token = (qs.get("token") or [None])[0]
    if token:
        _RECENT_PASSWORD_RESET_TOKENS[email.lower()] = token


def get_recent_password_reset_token(email: str) -> str | None:
    """E2E test_helpers から呼ばれる accessor。"""
    return _RECENT_PASSWORD_RESET_TOKENS.get(email.lower())
```

### Step 1: FeatureForm に geometry input モード切替を追加

`app/src/components/features/feature-form.tsx` に以下を組み込む（既存コードを破壊しない形で追加）:

```tsx
const [inputMode, setInputMode] = useState<"map" | "geojson">("map");
const [geojsonText, setGeojsonText] = useState("");

// ... 既存の地図用 state ...

// JSX 内
<Tabs value={inputMode} onValueChange={(v) => setInputMode(v as "map" | "geojson")}>
  <TabsList>
    <TabsTrigger value="map" data-testid="feature-form-mode-map">マップで描画</TabsTrigger>
    <TabsTrigger value="geojson" data-testid="feature-form-mode-geojson">GeoJSON 直接入力</TabsTrigger>
  </TabsList>
  <TabsContent value="map">
    {/* 既存の GeometryEditor */}
  </TabsContent>
  <TabsContent value="geojson">
    <Textarea
      value={geojsonText}
      onChange={(e) => {
        setGeojsonText(e.target.value);
        try {
          const parsed = JSON.parse(e.target.value);
          // 簡易バリデーション + 既存 geometry state に流す
          if (parsed.type && parsed.coordinates) {
            setGeometry(parsed);
          }
        } catch {
          // 入力途中で parse できないのは OK
        }
      }}
      placeholder='{"type":"Point","coordinates":[139.7,35.6]}'
      data-testid="feature-form-geometry-text"
    />
  </TabsContent>
</Tabs>
```
